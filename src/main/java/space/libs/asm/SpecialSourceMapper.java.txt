package space.libs.asm;

import com.google.common.io.Resources;
import net.md_5.specialsource.JarMapping;
import net.md_5.specialsource.repo.RuntimeRepo;
import org.apache.logging.log4j.LogManager;

import java.io.*;

@SuppressWarnings("UnstableApiUsage")
public class SpecialSourceMapper {

    public SpecialSourceMapper() {}

    public ModJarRemapper remapper;

    public byte[] map(byte[] bytes, String mapping_name) {
        LogManager.getLogger().info("Loading Custom Mappings: " + mapping_name);
        try {
            JarMapping mapping = new JarMapping();
            InputStream resourceStream = Resources.getResource(mapping_name).openStream();
            BufferedReader reader = new BufferedReader(new InputStreamReader(resourceStream));
            mapping.loadMappings(reader, null, null, false);
            this.remapper = new ModJarRemapper(mapping);
        } catch (Exception e) {
            throw new RuntimeException("Failed to load mappings: " + mapping_name + " ! " + e);
        }
        return remapper.remapClassFile(bytes, RuntimeRepo.getInstance());
    }
}
